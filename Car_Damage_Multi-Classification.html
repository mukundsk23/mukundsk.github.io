Build CNN Model¶

In [0]:
# example of a 3-block vgg style architecture
model = Sequential()
# 32 filters each of 3*3 size on the input image
# Activation function in first layer 'Relu'
model.add(Conv2D(32, (3, 3), activation='relu', kernel_initializer='he_uniform', padding='same', input_shape=input_shape))
model.add(Conv2D(32, (3, 3), activation='relu', kernel_initializer='he_uniform', padding='same'))
# Here, max pooling is done to reduce the image size to half and exract important features
model.add(MaxPooling2D((2, 2)))
model.add(Conv2D(64, (3, 3), activation='relu', kernel_initializer='he_uniform', padding='same'))
model.add(Conv2D(64, (3, 3), activation='relu', kernel_initializer='he_uniform', padding='same'))
model.add(MaxPooling2D((2, 2)))
model.add(Conv2D(128, (3, 3), activation='relu', kernel_initializer='he_uniform', padding='same'))
model.add(Conv2D(128, (3, 3), activation='relu', kernel_initializer='he_uniform', padding='same'))
model.add(MaxPooling2D((2, 2)))

# example output part of the model
model.add(Flatten())
# 128 neurons to be connected in this layer
model.add(Dense(128, activation='relu', kernel_initializer='he_uniform'))
# 5 neurons to be connected in the final layer as we have 5 classes to be predicted
# The softmax function in the last layer gives maximum probability for one neuron and minimal for the other
model.add(Dense(5, activation='softmax'))
model.summary()

# Function to test predictions on validation data
def execute_model(image):
  test_img = image                                      
  scale_percent = 25                                   # percent of original size
  width = int(test_img.shape[1] * scale_percent / 100) # Set image width 
  height = int(test_img.shape[0] * scale_percent / 100)# Set image height
  dim = (width, height)                                
  test_img = cv2.resize(test_img,(256,256))            # Resizing image to 256*256
  
  x_test = np.array(test_img)                         # Converting image to numpy array
  x_test = np.reshape(x_test,(1,256,256,3))           # Reshaping data
  out = model.predict(x_test)                         # Predicting output
  pred_label = np.argmax(out, axis=1)                 # The maximum value along the given axis(target labels) is returned
  
  plt.imshow(test_img)                                # Plotting the image
  for key in c.keys():                                # For loop of target labels  
        if pred_label[0] == key:                      
            print ('Assessment: {} damage to vehicle'.format(c[key])) # The target label is printed
