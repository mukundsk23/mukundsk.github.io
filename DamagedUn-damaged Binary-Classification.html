Build a CNN model¶

In [9]:
# Initializing the CNN
classifier = Sequential()

# Step 1 - Convolution
classifier.add(Conv2D(32, (3, 3), input_shape = (256, 256, 3), padding="same", activation="relu"))

# Step 2 - Pooling
classifier.add(MaxPooling2D(pool_size = (2,2)))

# Step 3 - Flatten
classifier.add(Flatten())

# Step 4 - Full Connection
classifier.add(Dense(units = 128, activation = 'relu'))
classifier.add(Dense(units = 1, activation = 'sigmoid'))

from keras.preprocessing import image

img_predict = image.img_to_array(img_pred)          # Converting 2D image to 1D array
img_predict = np.expand_dims(img_pred, axis = 0)

# Prediction on the image

  prediction = "Damaged"
